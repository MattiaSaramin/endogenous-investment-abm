%----------------------------------------------------------- boxes/misc
\usepackage{xcolor}
% mdframed rather than tcolorbox: tcolorbox pulls in the whole of TikZ/PGF,
% which is by far the most expensive thing this document could load and is
% the usual cause of a timeout on Overleaf's free tier. mdframed with
% framemethod=default draws the same boxes with no TikZ at all.
\usepackage{mdframed}
\mdfdefinestyle{declstyle}{%
  linecolor=gray!55, linewidth=0.5pt,
  backgroundcolor=gray!5,
  frametitlebackgroundcolor=gray!12,
  frametitlefont=\bfseries\small,
  innerleftmargin=8pt, innerrightmargin=8pt,
  innertopmargin=6pt, innerbottommargin=6pt}
\newenvironment{declbox}[1][]
  {\begin{mdframed}[style=declstyle,#1]}
  {\end{mdframed}}
